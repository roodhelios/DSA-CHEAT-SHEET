Module 1 — Introduction to DSA

What is a Data Structure?

A meaningful way of arranging and storing data for efficient utilization and processing. The right choice depends on the nature of data and operations required.

Linear vs Non-Linear

LinearNon-Linear
Sequential arrangementHierarchical arrangement
One predecessor/successorMultiple children/parents
Array, Stack, Queue, LLTree, Graph
Simpler traversalComplex traversal

ADT — Abstract Data Type

Defines what data is stored and what operations exist — hiding implementation (black box).

Bag ADT Example

class Bag:
  def __init__(self):    # create empty bag
    self.items = []

  def __len__(self):
    return len(self.items)

  def add(self, item):
    self.items.append(item)

  def contains(self, item):
    return item in self.items

Stack Implementation (Python)

class Stack:
  def __init__(self):
    self._data = []                   # internal list

  def is_empty(self): return len(self._data) == 0
  def __len__(self):  return len(self._data)

  def push(self, item):
    self._data.append(item)           # O(1)

  def pop(self):
    if self.is_empty(): raise Exception("Empty stack")
    return self._data.pop()           # O(1)

  def peek(self):
    if self.is_empty(): raise Exception("Empty stack")
    return self._data[-1]             # O(1) — no removal

Queue Implementation (Python)

from collections import deque

class Queue:
  def __init__(self): self._q = deque()
  def is_empty(self): return len(self._q)==0
  def enqueue(self, v): self._q.append(v)
  def dequeue(self):
    if self.is_empty():
      raise Exception("Empty queue")
    return self._q.popleft()         # O(1)

Postfix Evaluation (Python)

def eval_postfix(expr):
  s = []
  for tok in expr.split():
    if tok.lstrip('-').isdigit():
      s.append(int(tok))
    else:
      b, a = s.pop(), s.pop()
      if   tok=='+': s.append(a+b)
      elif tok=='-': s.append(a-b)
      elif tok=='*': s.append(a*b)
      elif tok=='/': s.append(a//b)
  return s[0]

# Example: "3 4 + 2 *" → 14
print(eval_postfix("3 4 + 2 *"))

SLL Node & Insert Front

class Node:
  def __init__(self, data):
    self.data = data
    self.next = None

def insert_front(head, data):
  new = Node(data)
  new.next = head   # link before changing head
  head = new
  return head       # new head

def insert_end(head, data):
  new = Node(data)
  if not head: return new
  cur = head
  while cur.next: cur = cur.next
  cur.next = new
  return head

SLL Delete & Traverse

def remove(head, target):
  cur, prev = head, None
  while cur:
    if cur.data == target:
      if not prev: head = cur.next
      else: prev.next = cur.next
      return head
    prev, cur = cur, cur.next
  return head  # not found

def traverse(head):
  cur = head
  while cur:
    print(cur.data, end=" → ")
    cur = cur.next
  print("None")

DLL Node Structure

class DNode:
  def __init__(self, data):
    self.data = data
    self.prev = None   # link to previous node
    self.next = None   # link to next node

def insert_front_DLL(head, data):
  new = DNode(data)
  new.next = head
  if head: head.prev = new   # set back-link
  return new               # new head

BST Insert & Search

class BSTNode:
  def __init__(self, v):
    self.val = v
    self.left = self.right = None

def insert(root, v):
  if not root: return BSTNode(v)
  if v < root.val:
    root.left  = insert(root.left, v)
  elif v > root.val:
    root.right = insert(root.right, v)
  return root

def search(root, target):
  if not root or root.val==target:
    return root
  if target < root.val:
    return search(root.left, target)
  return search(root.right, target)

BST Traversals

def inorder(root, res=[]):
  if root:
    inorder(root.left, res)
    res.append(root.val)   # LNR: sorted!
    inorder(root.right, res)
  return res

def preorder(root, res=[]):
  if root:
    res.append(root.val)   # NLR
    preorder(root.left, res)
    preorder(root.right, res)
  return res

def postorder(root, res=[]):
  if root:
    postorder(root.left, res)
    postorder(root.right, res)
    res.append(root.val)   # LRN
  return res

BFS Algorithm (Python)

from collections import deque

def bfs(graph, start):
  visited = set()
  queue   = deque([start])
  visited.add(start)
  order   = []

  while queue:
    v = queue.popleft()     # dequeue
    order.append(v)
    for w in graph[v]:
      if w not in visited:
        visited.add(w)
        queue.append(w)     # enqueue

  return order            # BFS order

# Time: O(V+E), Space: O(V)

DFS Algorithm (Python)

def dfs(graph, start):
  visited = set()
  stack   = [start]
  order   = []

  while stack:
    v = stack.pop()         # pop from stack
    if v not in visited:
      visited.add(v)
      order.append(v)
      for w in graph[v]:
        if w not in visited:
          stack.append(w)   # push neighbours

  return order            # DFS order

# Time: O(V+E), Space: O(V)

Dijkstra's Shortest Path (Python)

import heapq

def dijkstra(graph, src):
  # graph = {node: [(neighbour, weight), ...]}
  dist = {v: float('inf') for v in graph}
  dist[src] = 0
  pq = [(0, src)]          # (distance, node)

  while pq:
    d, u = heapq.heappop(pq)
    if d > dist[u]: continue  # stale entry
    for v, w in graph[u]:
      nd = dist[u] + w
      if nd < dist[v]:
        dist[v] = nd
        heapq.heappush(pq, (nd, v))
  return dist

# Example usage:
G = {'A':[('B',4),('C',2)], 'B':[('C',1),('D',5)],
     'C':[('B',1),('D',8)], 'D':[]}
print(dijkstra(G, 'A'))   # {'A':0,'B':3,'C':2,'D':8}

Kruskal's MST (Python)

def kruskal(edges, n):
  # edges = [(weight, u, v)]; n = number of nodes
  edges.sort()          # sort by weight O(E log E)
  parent = list(range(n))

  def find(x):          # union-find
    while parent[x] != x: parent[x] = parent[parent[x]]; x = parent[x]
    return x

  mst, cost = [], 0
  for w, u, v in edges:
    pu, pv = find(u), find(v)
    if pu != pv:          # no cycle → add edge
      parent[pu] = pv
      mst.append((u, v, w)); cost += w
  return mst, cost
